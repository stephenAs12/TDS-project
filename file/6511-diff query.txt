SELECT users.fname, users.mname, woreda.woreda_name
FROM users
INNER JOIN woreda
ON users.aname=woreda.woreda_id;

FROM users
INNER JOIN woreda
ON users.aname=woreda.woreda_id
INNER JOIN job_level
ON users.jlevel=job_level.level_id;
